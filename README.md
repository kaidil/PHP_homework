CREATE TABLE tree (

    tree_name varchar(255) NOT NULL,

);

CREATE TABLE fruit (

    fruit_name varchar(255) NOT NULL,

);

    insect_id int NOT NULL AUTO_INCREMENT UNIQUE,

    tree_fruit_id int NOT NULL AUTO_INCREMENT UNIQUE,

	tree_id INT,
	fruit_id INT,
	FOREIGN KEY (tree_id) REFERENCES tree(tree_id),
